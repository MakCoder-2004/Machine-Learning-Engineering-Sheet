# Python OOP (Object-Oriented Programming)

OOP is a programming paradigm that uses objects and classes to create programs.


## OOP Terms:
**Abstraction:** Hiding implementation details of a class from other programmers
**Encapsulation:** Restricting access to certain attributes and methods of a class from other programmers
**Inheritance:** Reusing code from other classes
**Polymorphism:** Using the same interface for different underlying forms (data types)

### Private and Public Instance Variables

**Public Instance Variables**
```python
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def multX(self):
        return self.x * self.x

    def multY(self):
        return self.y * self.y

# Usage
newPoint = Point(10, 20)

print(newPoint.x)
print(newPoint.y)

print(newPoint.multX())
print(newPoint.multY()) 
```

**Private Instance Variables**
```python
class Point:
    def __init__(self, x, y):
        self.__x = x  # private attribute
        self.__y = y  # private attribute
    
    def multX(self):
        return self.__x * self.__x
    
    def multY(self):
        return self.__y * self.__y

    # Optional: provide getters to access private attributes safely
    def getX(self):
        return self.__x
    
    def getY(self):
        return self.__y


# Usage
newPoint = Point(10, 20)

print(newPoint.getX())      # Access via getter
print(newPoint.__x)         # Access directly (Gives An Error)
print(newPoint._Point__x)   # Access via name mangling

print(newPoint.multX())
print(newPoint.multY())
```

### Converting Object To String (`__str__` Method)
```python

    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __str__(self):
        return f"({self.x}, {self.y})"
```

### Adding and Subtracting Instances
```python
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __str__(self):
        return f"({self.x}, {self.y})"
    
    def __add__(self, other):
        return Point(self.x + other.x, self.y + other.y)
    
    def __sub__(self, other):
        return Point(self.x - other.x, self.y - other.y)
    
p1 = Point(10, 20)
p2 = Point(30, 40)

p_add = p1 + p2
print(p_add)
p_sub = p1 - p2
print(p_sub)
```

### Sorting List of Instances
```python
class Fruit:
    def __init__(self, name, price):
        self.name = name
        self.price = price

fruits = [Fruit("Orange", 10), Fruit("Apple", 5), Fruit("Banana", 20)]

for fruit in sorted(fruits, key = lambda fruit: fruit.name):
    print(fruit.name)

```

**OR**

```python
class Fruit:
    def __init__(self, name, price):
        self.name = name
        self.price = price

    def sort_priority(self): return self.name

fruits = [Fruit("Orange", 10), Fruit("Apple", 5), Fruit("Banana", 20)]

for fruit in sorted(fruits, key = Fruit.sort_priority):
    print(fruit.name)
```

**OR**

```python
class Fruit:
```

## Inheritance

### Single Inheritance

```python
CURRENT_YEAR = 2019

class Person:
    def __init__(self, name, year_born):
        self.name = name
        self.year_born = year_born
    
    def getAge(self):
        return CURRENT_YEAR - self.year_born
    
    def __str__(self):
        return '{} ({})'.format(self.name, self.getAge())

class Student(Person):
    def __init__(self, name, year_born):
        Person.__init__(self, name, year_born) # Call the constructor of the parent class
        self.knowledge = 0
    
    def study(self):
        self.knowledge += 1

alice = Student('Alice Smith', 1990)
alice.study()
print(alice.knowledge)
```

### Multiple Inheritance

```python
class Swimmer:
    def swim(self):
        print("Swimming!")

class Flyer:
    def fly(self):
        print("Flying!")

class Duck(Swimmer, Flyer):
    pass

daffy = Duck()
daffy.swim()
daffy.fly()
```

### Overriding Methods

```python
from random import randrange

# Here's the original Pet class
class Pet():
    boredom_decrement = 4
    hunger_decrement = 6
    boredom_threshold = 5
    hunger_threshold = 10
    sounds = ['Mrrp']
    def __init__(self, name = "Kitty"):
        self.name = name
        self.hunger = randrange(self.hunger_threshold)
        self.boredom = randrange(self.boredom_threshold)
        self.sounds = self.sounds[:]  # copy the class attribute, so that when we make changes to it, we won't affect the other Pets in the class

    def clock_tick(self):
        self.boredom += 1
        self.hunger += 1

    def mood(self):
        if self.hunger <= self.hunger_threshold and self.boredom <= self.boredom_threshold:
            return "happy"
        elif self.hunger > self.hunger_threshold:
            return "hungry"
        else:
            return "bored"

    def __str__(self):
        state = "     I'm " + self.name + ". "
        state += " I feel " + self.mood() + ". "
        # state += "Hunger %d Boredom %d Words %s" % (self.hunger, self.boredom, self.sounds)
        return state

    def hi(self):
        print(self.sounds[randrange(len(self.sounds))])
        self.reduce_boredom()

    def teach(self, word):
        self.sounds.append(word)
        self.reduce_boredom()

    def feed(self):
        self.reduce_hunger()

    def reduce_hunger(self):
        self.hunger = max(0, self.hunger - self.hunger_decrement)

    def reduce_boredom(self):
        self.boredom = max(0, self.boredom - self.boredom_decrement)

```
